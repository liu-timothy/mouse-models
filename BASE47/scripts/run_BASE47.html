# Ensure matrix is numeric
RNAseq_raw_base47[] <- lapply(RNAseq_raw_base47, as.numeric)

# Convert cpm to log2 upper quartile-normalized expression (that's how they did it in their methods)
#   1. For each sample, calculate 75th percentile of nonzero CPM values
#   2. Divide each sample by its own upper-quartile value
#   3. Multiply by 1000 to put values on comparable scale
#   4. Apply log2(x+1) transformation
cpm_to_uq_log2 <- function(cpm_matrix) {
  uq <- apply(cpm_matrix, 2, function(x) quantile(x[x > 0], 0.75))
  scaled <- sweep(cpm_matrix, 2, uq, "/") * 1000
  log2(scaled + 1)
}

base47_uqlog2 <- RNAseq_raw_base47 %>% 
  cpm_to_uq_log2() %>% 
  rename_with(~ str_remove(., "_cpm"), ends_with("_cpm")) 

# median-center every gene by subtracting its row median from every sample (median-centers)
base47_input <- sweep(base47_uqlog2, 1, apply(base47_uqlog2, 1, median, na.rm = TRUE), "-")

# Every row median should now be exactly 0 (sanity check that matrix is converted correctly)
stopifnot(all(abs(apply(base47_input, 1, median, na.rm = TRUE)) < 1e-10))

#Create new columns for expected mol subtype and its BASE47 call
unc.tma.rna.calls_formatted <- unc.tma.rna.calls %>%
  rownames_to_column("Sample") %>%
  mutate(
    Tumor.Model = case_when(
      str_detect(Sample, "^BBN") ~ "BBN",
      str_detect(Sample, "^UPPL") ~ "UPPL",
      TRUE ~ NA_character_
    ),
    Basal.Score = Basal.Correlation - Luminal.Correlation
  ) %>%

        as.numeric(sub(".*?([0-9]+)$", "\\1", Sample))

    )
  )
